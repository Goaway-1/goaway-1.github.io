반복문

Jul 28, 2022 About 5 mins

#CodingTest#C++

문제

정수 n이 매개변수로 주어집니다. 다음 그림과 같이 밑변의 길이와 높이가 n인 삼각형에서 맨 위 꼭짓점부터 반시계 방향으로 달팽이 채우기를 진행한 후, 첫 행부터 마지막 행까지 모두 순서대로 합친 새로운 배열을 return 하도록 solution 함수를 완성해주세요.

조건

  • n은 1 이상 1,000 이하입니다.

입출력 예시

N result
4 [1,2,9,3,10,8,4,5,6,7]
5 [1,2,12,3,13,11,4,14,15,10,5,6,7,8,9]
6 [1,2,15,3,16,14,4,17,21,13,5,18,19,20,12,6,7,8,9,10,11]

풀이방식

나의 풀이방식

초기에는 반복문을 통해 하나씩 추가하고 싶지 않고, 특정 알고리즘(계산식)을 통한 해결방법을 구상했는데 마음대로 되지 않아서 포기했다.

이 계산 진행 방식은 아래와 같은 순서를 계속 반복한다.

  1. 아래로 내려가는 경우
  2. 오른쪽으로 움직이는 경우
  3. 위쪽으로 올라가는 경우
  vector<int> solution(int n) {
    vector<int> answer;
    vector<vector<int>> arr(n+1, vector<int>(n+1));
    int maxCnt = (n * (n+1)) / 2, cnt = 1;                //max는 종료 조건..!
    int top = 1, left = 1, bottom = n, right = 0;

    //방향
    string WayToGo = "Down";

    while (maxCnt >= cnt){
      if (WayToGo == "Down") {
        for (int i = top; i <= bottom; i++) arr[i][left] = cnt++;
        top++;
        left++;
        WayToGo = "Right";
      }
      else if (WayToGo == "Right") {
        for (int i = left; i <= bottom - right; i++) arr[bottom][i] = cnt++;
        bottom--;
        WayToGo = "Up";
      }
      else if (WayToGo == "Up") {
        for (int i = bottom; i >= top; i--) arr[i][i-right] = cnt++;
        right++;
        top++;
        WayToGo = "Down";
      }
    }
    for (int i = 1; i <= n; i++)for (int j = 1; j <= i; j++)answer.push_back(arr[i][j]);
    return answer;
  }

다른이의 풀이방식

동일하나 조금 더 직관적인 풀이 방식

  int arr[1000][1000];
  vector<int> solution(int n) {
    vector<int> answer;
    int p = n;
    int k = 1;
    int x = -1, y = 0;

    for (int i = 0; i < n; i++) {
      for (int j = p;j >= 1; j--) {
        if(i % 3 == 0) arr[++x][y] = k;
        else if (i % 3 == 1) arr[x][++y] = k;
        else if (i % 3 == 2) arr[--x][--y] = k;
        k++;
      }
      p--;
    }

    for (int i = 0; i < n; i++) for (int j = 0; j <= i;  j++) answer.emplace_back(arr[i][j]);

    return answer;
  }
