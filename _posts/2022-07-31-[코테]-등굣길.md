## 문제

  계속되는 폭우로 일부 지역이 물에 잠겼습니다. 물에 잠기지 않은 지역을 통해 학교를 가려고 합니다. 집에서 학교까지 가는 길은 m x n 크기의 격자모양으로 나타낼 수 있습니다.
  
  가장 왼쪽 위, 즉 집이 있는 곳의 좌표는 (1, 1)로 나타내고 가장 오른쪽 아래, 즉 학교가 있는 곳의 좌표는 (m, n)으로 나타냅니다.

  격자의 크기 m, n과 물이 잠긴 지역의 좌표를 담은 2차원 배열 puddles이 매개변수로 주어집니다. 오른쪽과 아래쪽으로만 움직여 집에서 학교까지 갈 수 있는 최단경로의 개수를 1,000,000,007로 나눈 나머지를 return 하도록 solution 함수를 작성해주세요.

## 조건

  - 격자의 크기 m, n은 1 이상 100 이하인 자연수입니다.
    - m과 n이 모두 1인 경우는 입력으로 주어지지 않습니다.
  - 물에 잠긴 지역은 0개 이상 10개 이하입니다.
  - 집과 학교가 물에 잠긴 경우는 입력으로 주어지지 않습니다.

## 입출력 예시

  |m|n|puddles|return|
  |:--:|:--:|:--:|:--:|
  |4|3|[[2,2]]|4|

## 풀이방식

### 1번 풀이방식

  > 효율성 만족 X

  그냥 BFS로 풀었는데 방법도 더럽고, 효율성 테스트를 통과하지 못함

  ```cpp
  #include <queue> 

  int solution(int m, int n, vector<vector<int>> puddles) {
    int answer = 0;
    vector<vector<pair<int,int>>> arr(m+1,vector<pair<int,int>>(n+1));      //최소 거리, 개수
    priority_queue<pair<int,int>> Q;

    // can't to go
    for (int i = 0; i < puddles.size(); i++) arr[puddles[i][0]][puddles[i][1]] = make_pair(-1,0);

    Q.push(make_pair(1,1));
    arr[1][1] = make_pair(0,1);
    while (!Q.empty()) {
      int x = Q.top().first;
      int y = Q.top().second;
      Q.pop();

      // loop
      if (x+1 <= m && arr[x + 1][y].first != -1) {
        if (arr[x + 1][y].first == 0 || arr[x + 1][y].first > arr[x][y].first + 1) {
          if(arr[x + 1][y].first == 0) Q.push(make_pair(x + 1, y));
          arr[x + 1][y] = make_pair(arr[x][y].first + 1, arr[x][y].second);
        }
        else if(arr[x + 1][y].first == arr[x][y].first + 1){
          if (y-1 > 0) {
            arr[x + 1][y].second = arr[x][y].second + arr[x + 1][y-1].second;
            Q.push(make_pair(x + 1, y));
          }
          else arr[x + 1][y].second = arr[x][y].second;
        }
      }
      if(y + 1 <= n && arr[x][y+1].first != -1) {
        if (arr[x][y + 1].first == 0 || arr[x][y + 1].first > arr[x][y].first + 1) {
          if(arr[x][y + 1].first == 0) Q.push(make_pair(x, y + 1));
          arr[x][y + 1] = make_pair(arr[x][y].first + 1, arr[x][y].second);
        }
        else if (arr[x][y + 1].first == arr[x][y].first + 1) {
          if (x - 1 > 0) {
            arr[x][y + 1].second = arr[x][y].second + arr[x - 1][y + 1].second;
            Q.push(make_pair(x, y + 1));
          }
          else arr[x][y + 1].second = arr[x][y].second;
        }
      }
    }
    answer = arr[m][n].second;
    return answer;
  }
  ```

### 2번 풀이방식

  > 모두 만족

  전에 인프런에서 풀었던 사과나무 방식의 문제가 생각나서 풀이를 진행했으며, 이전 값들에 의해 현재의 값이 정해지는 방식이다. 이 풀이에서는 (n,m)의 값은 (n-1,m), (n, m-1)의 값에 의해 영향을 받는다. 그래서 x 또는 y가 1일때만 먼저 처리하여 계산값을 도출하는 것이다. 자세한 계산 방식은 아래에서 설명한다.

  <img src="https://raw.githubusercontent.com/Goaway-1/goaway-1.github.io/master/_posts/images/gotoschool.png" height="350" title="gotoschool">

  이때 arr배열은 pair로 되어 있으며, __"최소 거리, 최소 거리인 경로의 개수"로__ 구성되어 있다. 아래는 로직이다.

  1. 물웅덩이로 갈 수 없는 곳을 (-1,0)으로 채움
  2. x = 1 or y = 1일때의 최소 비용과 경로의 개수를 먼저 채움
  3. arr의 (2,2) 부터 (n-1,m)과 (n,m-1)의 값(즉 위와 좌측의 값)을 사용하여 최소 비용과 경로의 개수를 계산
    - __이전 값의 최소 거리가 모두 -1이라면__ (-1,0) 삽입
    - __(n-1,m)의 최소 거리가 -1이거나, (n,m-1)보다 크다면__ ((n,m-1)의 최소 거리, (n,m-1)의 경로 개수) 삽입
    - __(n,m-1)의 최소 거리가 -1이거나, (n-1,m)보다 크다면__ ((n-1,m)의 최소 거리, (n-1,m)의 경로 개수) 삽입
    - __이전 값들의 최소 거리가 동일하다면__ ((n,m-1)의 최소 거리, (n,m-1) + (n-1,m)의 경로 개수) 삽입

  ```cpp
  int solution(int m, int n, vector<vector<int>> puddles) {
    const int tmp = 1000000007;
    vector<vector<pair<int,int>>> arr(m+1,vector<pair<int,int>>(n+1));      //최소 거리, 경로 개수

    // 1번 : 갈 수 없는 곳을 선처리
    arr[1][1] = make_pair(0, 1);
    for (int i = 0; i < puddles.size(); i++) arr[puddles[i][0]][puddles[i][1]] = make_pair(-1, 0);

    /* 2번 : x = 1 or y = 1인 경우만 계산
    *  이전 값이 -1이라면 다음 배열부터는 무조건 (-1,0)으로 구성
    */
    bool isbreak = false;
    for (int i = 2; i <= n; i++) {
      if (isbreak) {
        arr[1][i] = make_pair(-1,0);
        continue;
      }

      if(arr[1][i].first != -1) arr[1][i] = make_pair(arr[1][i - 1].first + 1 , 1);
      else {
        arr[1][i] = make_pair(-1, 0);
        isbreak = true;
      }
    }

    isbreak = false;
    for (int i = 2; i <= m; i++) {
      if (isbreak) {
        arr[i][1] = make_pair(-1, 0);
        continue;
      }

      if (arr[i][1].first != -1) arr[i][1] = make_pair(arr[i-1][1].first + 1, 1);
      else {
        arr[i][1] = make_pair(-1, 0);
        isbreak = true;
      }
    }

    // 3번 : Main Logic
    for (int i = 2; i <= n; i++) {
      for (int j = 2; j <= m; j++) {
        if(arr[j][i].first == -1) continue; 

        if (arr[j - 1][i].first == -1 && arr[j][i - 1].first == -1) {
          arr[j][i] = make_pair(-1, 0);
          continue;
        }
            
        if (arr[j - 1][i].first != arr[j][i - 1].first) {
          if (arr[j - 1][i].first == -1) arr[j][i] = make_pair(arr[j][i - 1].first + 1, arr[j][i - 1].second % tmp);
          else if (arr[j][i - 1].first == -1) arr[j][i] = make_pair(arr[j - 1][i].first + 1, arr[j - 1][i].second % tmp);
          else {
            if (arr[j - 1][i].first > arr[j][i - 1].first) arr[j][i] = make_pair(arr[j][i - 1].first + 1, arr[j][i - 1].second % tmp);
            else arr[j][i] = make_pair(arr[j - 1][i].first + 1, arr[j - 1][i].second % tmp);
          }       
          }
          else  arr[j][i] = make_pair(arr[j - 1][i].first + 1, (arr[j - 1][i].second + arr[j][i - 1].second) % tmp);
      }
    }
    return arr[m][n].second % tmp;
  }
  ```
  
### 다른이의 풀이방식

  > 최소거리를 비교하지 않음

  ```cpp
  int solution(int m, int n, vector<vector<int>> puddles) {
    int loc[m][n];

    // init arr
    for (int i = 0; i < m; ++i) for (int j = 0; j < n; ++j) loc[i][j] = 0;

    // 1번
    for (vector<int> p : puddles) loc[p[0]-1][p[1]-1] = -1;

    for (int i = 0; i < m; ++i){
      if (loc[i][0] == -1) break;
      loc[i][0] = 1;
    }

    for (int i = 1; i < n; ++i){
      if (loc[0][i] == -1) break;
      loc[0][i] = 1;
    }

    for (int i = 1; i < m; ++i){
      for (int j = 1; j < n; ++j){
        if (loc[i][j])continue;

        if (loc[i-1][j] != -1) loc[i][j] += loc[i-1][j];
        if (loc[i][j-1] != -1) loc[i][j] += loc[i][j-1];

        loc[i][j] %= 1000000007;
      }
    }
    return loc[m-1][n-1];
  }
  ```
