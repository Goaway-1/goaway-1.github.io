## 문제

  N×N크기의 땅이 있고, 땅은 1×1개의 칸으로 나누어져 있다. 각각의 땅에는 나라가 하나씩 존재하며, r행 c열에 있는 나라에는 A[r][c]명이 살고 있다. 인접한 나라 사이에는 국경선이 존재한다. 모든 나라는 1×1 크기이기 때문에, 모든 국경선은 정사각형 형태이다.

  오늘부터 인구 이동이 시작되는 날이다.

  인구 이동은 하루 동안 다음과 같이 진행되고, 더 이상 아래 방법에 의해 인구 이동이 없을 때까지 지속된다.

  - 국경선을 공유하는 두 나라의 인구 차이가 L명 이상, R명 이하라면, 두 나라가 공유하는 국경선을 오늘 하루 동안 연다.
  - 위의 조건에 의해 열어야하는 국경선이 모두 열렸다면, 인구 이동을 시작한다.
  - 국경선이 열려있어 인접한 칸만을 이용해 이동할 수 있으면, 그 나라를 오늘 하루 동안은 연합이라고 한다.
  - 연합을 이루고 있는 각 칸의 인구수는 (연합의 인구수) / (연합을 이루고 있는 칸의 개수)가 된다. 편의상 소수점은 버린다.
  - 연합을 해체하고, 모든 국경선을 닫는다.
  - 각 나라의 인구수가 주어졌을 때, 인구 이동이 며칠 동안 발생하는지 구하는 프로그램을 작성하시오.

  [링크](https://www.acmicpc.net/problem/16234)
  
## 조건

  - 첫째 줄에 N, L, R이 주어진다. (1 ≤ N ≤ 50, 1 ≤ L ≤ R ≤ 100)
  - 둘째 줄부터 N개의 줄에 각 나라의 인구수가 주어진다. r행 c열에 주어지는 정수는 A[r][c]의 값이다. (0 ≤ A[r][c] ≤ 100)
  - 인구 이동이 발생하는 일수가 2,000번 보다 작거나 같은 입력만 주어진다.

## 입출력 예시
  
  |N|L|R|Arr|answer|
  |:--:|:--:|:--:|:--:|:--:|
  |2|20|50|[[50,30],[20,40]]|1|
  |2|40|50|[[50,30],[20,40]]|0|
  |3|5|10|[[10,15,20],[20,30,25],[40,22,10]]|2|
  |4|10|50|[[10,100,20,90],[80,100,60,70],[70,20,30,40],[50,20,100,10]]|3|

## 풀이방식

### 나의 풀이방식

  > BFS 풀이로 더 이상 움직일 수 없을때까지 반복하기에 조금 복잡하다.

  복잡해 보이지만 BFS를 가능할때까지 반복하는 로직이다. 시간복잡도가 n^5로 높지만 땅(N)의 넓이나, 인구 이동 발생횟수가 2000번보다 작거나 같게 되어 있기 때문에 그냥 진행하였다.

  아마 의도한듯 하다.

  ```cpp
  #include <iostream>
  #include <vector>
  #include <algorithm>
  #include <queue>
  using namespace std;

  int arr[101][101];
  int way[4][2] = [[0, 1], [1, 0], [0, -1], [-1, 0]];  // {}로 수정 필요! (깃블로그 오류) 

  int main() {
    int N = 0, L = 0, R = 0, ans = 0;
    cin >> N >> L >> R;

    // INIT
    for (int i = 0; i < N; i++) for (int j = 0; j < N; j++) cin >> arr[i][j];

    while (true) {
      bool isDone = false;
      vector<vector<int>> ch(N + 1, vector<int>(N + 1,0));    //해당 국가가 이미 그룹을 이루었는지 여부
      for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {

          if (ch[i][j] == 0) {
            queue<pair<int,int>> Q;         // BFS를 위한 Queue
            vector<pair<int, int>> vec;			// 그룹을 이룬 국가
            int val = 0;						        // 국가 수의 합
            ch[i][j] = 1;
            Q.push(make_pair(i,j));
            vec.push_back(make_pair(i,j));

            // BFS
            while (!Q.empty()) {
              int x = Q.front().first;        
              int y = Q.front().second;    
              val += arr[x][y];
              Q.pop();

              for (int i = 0; i < 4; i++) {
                int xx = x + way[i][0];
                int yy = y + way[i][1];
                int tmp = abs(arr[x][y] - arr[xx][yy]);

                if (xx >= 0 && xx < N && yy >= 0 && yy < N && ch[xx][yy] == 0 && tmp >= L && tmp <= R) {
                  isDone = true;
                  vec.push_back(make_pair(xx,yy));
                  Q.push(make_pair(xx,yy));
                  ch[xx][yy] = 1;
                }
              }
            }

            // 팀을 이룬 국가들의 인구이동
            if (vec.size() > 1) {
              val = val / vec.size();
              for (int h = 0; h < vec.size(); h++)arr[vec[h].first][vec[h].second] = val;
            }
          }
        }
      }

      if (!isDone) break;
      else ans++;
    }

    cout << ans;
    return 0;
  }
  ```

### 다른이의 풀이

  > 역시 BFS를 사용한 풀이

  시간복잡도가 가장 낮은 다른 사람의 풀이를 확인했는데, 별 다름이 없고, 방문한 국가의 확인(vis)을 재사용함에 따른 속도차이가 존재했다.

  ```cpp
  #include <bits/stdc++.h>
  using namespace std;
  #define X first
  #define Y second

  int n, l, r;
  int dx[4] = {1, 0, -1, 0};
  int dy[4] = {0, 1, 0, -1};
  int board[105][105];
  int vis[105][105];
  int ans = 0;

  int main(void){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n >> l >> r;

    // INIT
    for(int i = 0; i < n; i++) for(int j = 0; j < n; j++) cin >> board[i][j];

    while(1){
      bool flag = false;
      for(int i = 0; i < 105; i++) fill(vis[i], vis[i]+105, 0);
      queue<pair<int, int>> q;

      for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){

          if(!vis[i][j]) q.push({i, j});
          int sum = 0;
          vector<pair<int, int>> tmp;

          while(!q.empty()){
            auto cur = q.front(); q.pop();

            for(int dir = 0; dir < 4; dir++){
              int nx = cur.X + dx[dir];
              int ny = cur.Y + dy[dir];
              if(nx < 0 || nx >= n || ny < 0 || ny >= n) continue;
              int diff = abs(board[cur.X][cur.Y] - board[nx][ny]);

              if(diff >= l && diff <= r){
                flag = 1;
                if(!vis[cur.X][cur.Y]){
                  tmp.push_back({cur.X, cur.Y});
                  sum += board[cur.X][cur.Y];
                  vis[cur.X][cur.Y] = 1;
                }
                if(!vis[nx][ny]){
                  tmp.push_back({nx, ny});
                  sum += board[nx][ny];
                  vis[nx][ny] = 1;
                  q.push({nx,ny});
                }
              }
            }
          }

          for(int k = 0; k < tmp.size(); k++) board[tmp[k].X][tmp[k].Y] = sum/tmp.size();
        }
      }
      if(!flag) break;
      ans++;
    }
    cout << ans;
  }
  ```
