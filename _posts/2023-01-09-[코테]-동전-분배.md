## 문제

  윤화와 준희는 솔선수범하여 쓰레기를 줍는 착한 일을 하였다. 원장선생님께서는 윤화와 준희를 칭찬하시고 과자나 사 먹으라고 하시며 동전 몇 개를 윤화와 준희에게 건네 주었다.

  그런데 돈을 받은 윤화와 준희는 좋아하기보다 고민에 빠지고 말았다. 원장선생님께 받은 이 돈을 어떻게 나누어 할지 고민에 빠진 것이다. 두 사람 모두 상대방이 자기보다 1원이라도 더 받는 것은 도저히 인정할 수 없어 한다. 따라서 돈을 똑같이 둘로 나누어 가져야 두 사람이 모두 만족할 수 있게 된다.

  하지만 두 사람에게 돈을 똑같이 나누는 것이 불가능한 경우도 있다. 예를 들어 500원짜리 1개와 50원짜리 1개를 받았다면, 이 돈을 두 사람이 똑같이 나누어 가질 수는 없다. 물론 동전을 반으로 잘라서 나누어 가질 수도 있겠지만 그러면 돈으로서의 가치를 잃기 때문에 그렇게 할 수는 없다.

  이제 우리가 할 일은 다음과 같다. 원장 선생님께서 N가지 종류의 동전을 각각 몇 개씩 주셨을 때, 그 돈을 반으로 나눌 수 있는지 없는지 판단하는 것이다.

  [링크](https://www.acmicpc.net/problem/1943)
  
## 조건

  - 세 개의 입력이 주어진다. 각 입력의 첫째 줄에 동전의 종류 N(1 ≤ N ≤ 100)이 주어진다. 
    - 각 입력의 둘째 줄부터 N+1째 줄까지 각각의 동전의 금액과 개수가 빈 칸을 사이에 두고 주어진다. 
    - 단, 원장선생님께서 주신 금액의 총 합은 100,000원을 넘지 않는다. 동전의 금액과 개수는 자연수이고, 같은 금액을 가진 동전이 두 번 이상 주어지는 경우는 없다.

## 입출력 예시
  
  |N|arr|answer|
  |:--:|:--:|:--:|
  |2|[[500,1],[50,1]]|0|
  |3|[[100,2],[50,1],[10,5]]|1|
  |3|[[1,1],[2,1],[3,1]]|1|

## 풀이방식

### 1번 풀이방식

  > DFS를 활용한 방식
  
  정답은 도출되지만 보기 좋게 타임 아웃이 난 사례...

  ```cpp
  #include<iostream>
  #include<string>
  #include<vector>
  #include<algorithm>
  using namespace std;

  int arr[100001];
  bool isDone = false;

  void DFS(const int targetVal, vector<int> &coins,int idx,int curVal) {
    if (isDone || curVal > targetVal) return;
    
    if (curVal == targetVal)  isDone = true;
    else {
      for (int i = idx; i < coins.size(); i++){
        if (arr[coins[i]] >= 1) {
          int cnt = 0;
          int tmp = arr[coins[i]];
          while (tmp != 0) {
            tmp--;
            cnt++;
            DFS(targetVal, coins, idx + 1,curVal + (coins[i] * cnt));
            if (isDone) return;
          }
          DFS(targetVal, coins, idx + 1, curVal);
        }
      }
    }
  }

  int main(void){
    int N = 0, coin = 0, count = 0, sum = 0;

    // INIT
    cin >> N;
    vector<int> coins;
	  for (int i = 0; i < N; i++) {
      cin >> coin >> count;
      coins.push_back(coin);
      arr[coin] = count;
      sum += (coin * count);
    }

    // 반의 반 & 순서대로 정렬
    sort(coins.begin(), coins.end());
    
    if(sum % 2 == 0) {
      sum /= 2;
      DFS(sum, coins,0,0);
    }

    if(isDone) cout << "1\n";
    else cout << "0\n";
  }
  ```


### 2번 풀이방식

  > DP를 활용한 방식
  
  가장 큰 수부터 현재 돈으로 만들 수 있는 경우의 dp[]를 1로 만든다.

  그 후 작은 수와 사전에 진행한 큰 수의 합을 통하여 dp[]를 1로 만든다. 이 과정을 모두 완료한 후에 sum의 절반 인덱스가 1인지 확인하고, 그렇다면 1을 출력한다.

  ```cpp
  #include<iostream>
  #include<vector>
  #include<algorithm>
  using namespace std;
  int dp[50001];

  int main() {
    int t = 3, n = 0, coin = 0, cnt = 0;
    dp[0] = 1;

    while (t--) {
      cin >> n;
      int sum = 0;
      vector <pair<int, int>> vec;

      for (int i = 0; i < n; i++) {
        cin >> coin >> cnt;
        vec.push_back(make_pair(coin, cnt));
        sum += coin * cnt;
      }

      // sum이 나누어 떨어지지 않는 경우
      if (sum % 2 == 1) {
        cout << 0 << '\n';
        continue;
      }

      // DP (top-down) 방식..
      // 현재 돈으로 만들 수 있는 경우의 수를 모두 추가...
      for (int i = 0; i < n; i++){
        for (int j = 50000; j >= vec[i].first; j--) {
          if (dp[j - vec[i].first] == 1) {
            for (int k = 0; k < vec[i].second && (j + k * vec[i].first) <= 50000; k++) {
              dp[j + k * vec[i].first] = 1;
            }
          }
        }
      }

      if (dp[sum / 2] == 1) cout << 1 << '\n';
      else cout << 0 << '\n';

      // 배열 정보 리셋
      memset(dp, 0, sizeof(dp));
      dp[0] = 1;
    }
    return 0;
  }
  ```
