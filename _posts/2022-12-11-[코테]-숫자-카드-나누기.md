## 문제

철수와 영희는 선생님으로부터 숫자가 하나씩 적힌 카드들을 절반씩 나눠서 가진 후, 다음 두 조건 중 하나를 만족하는 가장 큰 양의 정수 a의 값을 구하려고 합니다.

철수가 가진 카드들에 적힌 모든 숫자를 나눌 수 있고 영희가 가진 카드들에 적힌 모든 숫자들 중 하나도 나눌 수 없는 양의 정수 a
영희가 가진 카드들에 적힌 모든 숫자를 나눌 수 있고, 철수가 가진 카드들에 적힌 모든 숫자들 중 하나도 나눌 수 없는 양의 정수 a
예를 들어, 카드들에 10, 5, 20, 17이 적혀 있는 경우에 대해 생각해 봅시다. 만약, 철수가 [10, 17]이 적힌 카드를 갖고, 영희가 [5, 20]이 적힌 카드를 갖는다면 두 조건 중 하나를 만족하는 양의 정수 a는 존재하지 않습니다. 하지만, 철수가 [10, 20]이 적힌 카드를 갖고, 영희가 [5, 17]이 적힌 카드를 갖는다면, 철수가 가진 카드들의 숫자는 모두 10으로 나눌 수 있고, 영희가 가진 카드들의 숫자는 모두 10으로 나눌 수 없습니다. 따라서 철수와 영희는 각각 [10, 20]이 적힌 카드, [5, 17]이 적힌 카드로 나눠 가졌다면 조건에 해당하는 양의 정수 a는 10이 됩니다.

철수가 가진 카드에 적힌 숫자들을 나타내는 정수 배열 arrayA와 영희가 가진 카드에 적힌 숫자들을 나타내는 정수 배열 arrayB가 주어졌을 때, 주어진 조건을 만족하는 가장 큰 양의 정수 a를 return하도록 solution 함수를 완성해 주세요. 만약, 조건을 만족하는 a가 없다면, 0을 return 해 주세요.

[링크](https://school.programmers.co.kr/learn/courses/30/lessons/135807)

## 조건

- 1 ≤ arrayA의 길이 = arrayB의 길이 ≤ 500,000
- 1 ≤ arrayA의 원소, arrayB의 원소 ≤ 100,000,000
- arrayA와 arrayB에는 중복된 원소가 있을 수 있습니다.

## 입출력 예시

|arrayA|arrayB|result|
|:--:|:--:|:--:|
|[10, 17]|[5, 20]|0|
|[10, 20]|[5, 17]|10|
|[14, 35, 119]|[18, 30, 102]|7|

## 풀이방식

### 틀린 풀이 방식

> arrayA/B 원소의 공통 약수들 모든 약수를 구하여 비교한다.

```cpp
int arrA[100000001];
int arrB[100000001];         // 모든 약수 모음

int solution(vector<int> arrayA, vector<int> arrayB) {
    int answer = 0;
    vector<int> ansA, ansB;

    int maxA = *min_element(arrayA.begin(), arrayA.end());
    int maxB = *min_element(arrayB.begin(), arrayB.end());

    // arrayA의 공통 약수
    for (int i = 2; i <= maxA; i++) {
        bool isDone = true;
        for (int j = 0; j < arrayA.size(); j++){
            if (arrayA[j] % i != 0) {
                isDone = false;
                break;
            }
        }

        if (isDone) ansA.push_back(i);
    }

    // arrayA의 모든 약수
    sort(arrayA.begin(), arrayA.end());
    for (int i = 0; i < arrayA.size(); i++) {
        int tmp = arrayA[i];
        for (int j = tmp; j >= 2; j--){
            if (tmp % j == 0) {
                if (arrA[j] == 0) arrA[j] = 1;
                else break;
            }
        }
    }

    // arrayB의 공통 약수
    for (int i = 2; i <= maxB; i++) {
        bool isDone = true;
        for (int j = 0; j < arrayB.size(); j++) {
            if (arrayB[j] % i != 0) {
                isDone = false;
                break;
            }
        }
        if (isDone) ansB.push_back(i);
    }
        
    //arrayB의 모든 약수
    sort(arrayB.begin(), arrayB.end());
    for (int i = 0; i < arrayB.size(); i++) {
        int tmp = arrayB[i];
        for (int j = tmp; j >= 2; j--) {
            if (tmp % j == 0) {
                if (arrB[j] == 0) arrB[j] = 1;
                else break;
            }
        }
    }

    int idxA = ansA.size() - 1, idxB = ansB.size() - 1;
    while (idxA >= 0 || idxB >= 0) {
        if (idxA == -1) {
            if (arrA[ansB[idxB]] == 0) {
                answer = ansB[idxB];
                break;
            }
            idxB--;
        }
        else if (idxB == -1) {
            if (arrB[ansA[idxA]] == 0) {
                answer = ansA[idxA];
                break;
            }
            idxA--;
        }
        else {
            if (ansA[idxA] == ansB[idxB]) {         //같은 경우
                idxA--;
                idxB--;
            }
            else if (ansA[idxA] > ansB[idxB]) {     //A의 값이 더 큰 경우
                if (arrB[ansA[idxA]] == 0) {
                    answer = ansA[idxA];
                    break;
                }
                idxA--;
            }
            else {  //B의 값이 더 큰 경우
                if (arrA[ansB[idxB]] == 0) {
                    answer = ansB[idxB];
                    break;
                }
                idxB--;
            }
        }
    }

    return answer;
}
```

### 옳은 풀이 방식

> 각 ArrayA/B의 최대 공약수의 약수들을 구하고, 이 약수들로 ArrayA/B가 나누어지는지 확인

최대 공약수로만 ArrayA/B를 비교했을때는 모든 경우의 수를 파악할 수 없기 때문에 __"최대 공약수의 약수"들을 구하여 각 배열과 나누어지는지 확인하면__ 완료!

```cpp
int solution(vector<int> arrayA, vector<int> arrayB) {
    int answer = 0, ansA = 0, ansB = 0;
    vector<int> ansAA, ansBB;
        
    int maxA = *min_element(arrayA.begin(), arrayA.end());
    int maxB = *min_element(arrayB.begin(), arrayB.end());

    // arrayA와 arrayB의 최대 공약수
    for (int i = maxA; i >= 1; i--) {
        bool isDone = true;
        for (int j = 0; j < arrayA.size(); j++){
            if (arrayA[j] % i != 0) {
                isDone = false;
                break;
            }
        }

        if (isDone) {
            ansA = i;
            break;
        }
    }
    for (int i = maxB; i >= 1; i--) {
        bool isDone = true;
        for (int j = 0; j < arrayB.size(); j++) {
            if (arrayB[j] % i != 0) {
                isDone = false;
                break;
            }
        }

        if (isDone) {
            ansB = i;
            break;
        }
    }

    // 각 최대 공약수의 약수들...
    for (int i = 2; i < ansA + 1; i++){
        if (ansA % i == 0) ansAA.push_back(i);
    }
    for (int i = 2; i < ansB + 1; i++) {
        if (ansB % i == 0) ansBB.push_back(i);
    }

    // 각 최대 공약수의 약수로 나누어지는지 비교
    for (int i = 0; i < ansBB.size(); i++){
        bool isDone = true;
        for (int j = 0; j < arrayA.size(); j++){
            if (arrayA[j] % ansBB[i] == 0) {
                isDone = false;
                break;
            }
        }
        if (isDone) answer = max(answer, ansBB[i]);
    }
    for (int i = 0; i < ansAA.size(); i++) {
        bool isDone = true;
        for (int j = 0; j < arrayB.size(); j++) {
            if (arrayB[j] % ansAA[i] == 0) {
                isDone = false;
                 break;
            }
        }
        if (isDone) answer = max(answer, ansAA[i]);
    }

    return answer;
}
```
